    36    import xdc.rov.ViewInfo;
    37    
    38    /*!
    39     *  ======== Interrupt ======== 
    40     *  TCI663x based interrupt manager
    41     */
    42    @ModuleStartup
    43    
    44    module Interrupt inherits ti.sdo.ipc.notifyDrivers.IInterrupt
    45    {
    46        /* @_nodoc */
    47        metaonly struct InterruptDataStruct {
    48            UInt    remoteCoreId;
    49            String  isrFxn;
    50            String  isrArg;
    51            Bool    isFlagged;
    52        }
    53        
    54        /*!
    55         *  ======== rovViewInfo ========
    56         */
    57        @Facet
    58        metaonly config ViewInfo.Instance rovViewInfo =
    59            ViewInfo.create({
    60                viewMap: [
    61                    ['Registered Interrupts',
    62                        {
    63                            type: ViewInfo.MODULE_DATA,
    64                            viewInitFxn: 'viewInterruptsData',
    65                            structName: 'InterruptDataStruct'
    66                        }
    67                    ],
    68                ]
    69            });
    70    
    71    internal:
    72    
    73        /*! Shift value used for setting/identifying interrupt source */
    74        const UInt SRCSx_SHIFT = 4;
    75        
    76        /*! Ptr to the IPC Generation Registers */
    77        config Ptr IPCGR0;
    78    
    79        /*! Ptr to the IPC Acknowledgment Registers */
    80        config Ptr IPCAR0;
    81        
    82        /*! Inter-processor interrupt id (varies per device) */
    83        config UInt INTERDSPINT;
    84    
    85        /*!
    86         *  ======== intShmStub ========
    87         *  Stub function plugged as interrupt handler
    88         */
    89        Void intShmStub(UArg arg);
    90    
    91        struct Module_State {
    92            Fxn     func;
    93            UArg    args[];     /* One entry for each core */
    94            UInt    numPlugged; /* # of times the interrupt was registered */
    95        };
    96    }
