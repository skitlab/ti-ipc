    40    import ti.sdo.utils.MultiProc;
    41    
    42    /*!
    43     *  ======== InterruptHost ======== 
    44     *  TI81xx/A8 interrupt manager
    45     */
    46    
    47    module InterruptHost inherits ti.sdo.ipc.notifyDrivers.IInterrupt
    48    {
    49        /*! @_nodoc */
    50        metaonly struct InterruptDataView {
    51            String      remoteProcName;
    52            Bool        registered;
    53            Bool        enabled;
    54            Bool        intPending;
    55            Ptr         payload;
    56        };
    57        
    58        /*! @_nodoc */
    59        @Facet
    60        metaonly config xdc.rov.ViewInfo.Instance rovViewInfo = 
    61            xdc.rov.ViewInfo.create({
    62                viewMap: [
    63                    ['IncomingInterrupts',
    64                        {
    65                            type: xdc.rov.ViewInfo.MODULE_DATA,
    66                            viewInitFxn: 'viewInitInterrupt',
    67                            structName: 'InterruptDataView'
    68                        }
    69                    ],
    70                ]
    71            });
    72        
    73        /* Base address for the Mailbox subsystem */
    74        config UInt32 mailboxBaseAddr = 0x480C8000;
    75        
    76    internal:
    77    
    78        config UInt dspProcId   = MultiProc.INVALIDID;
    79        config UInt hostProcId  = MultiProc.INVALIDID;
    80        config UInt videoProcId = MultiProc.INVALIDID;
    81        config UInt vpssProcId   = MultiProc.INVALIDID;
    82    
    83        /*! Function table */
    84        struct FxnTable {
    85            Fxn    func;
    86            UArg   arg;
    87        }
    88    
    89        /*!
    90         *  ======== intShmStub ========
    91         *  Stub to be plugged
    92         */
    93        Void intShmStub(UArg arg);
    94        
    95        struct Module_State {        
    96            FxnTable   fxnTable[3];
    97            UInt       numPlugged;  /* # of times the interrupt was registered */
    98        };
    99    }
