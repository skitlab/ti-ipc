    40    import ti.sdo.utils.MultiProc;
    41    
    42    /*!
    43     *  ======== NotifySetup ======== 
    44     *  Notify setup proxy for Vayu
    45     *
    46     *  This module creates and registers all drivers necessary for inter-processor
    47     *  notification on Vayu. 
    48     */
    49    
    50    module NotifySetup inherits ti.sdo.ipc.interfaces.INotifySetup
    51    {
    52        /* Total number of cores on Vayu SoC */
    53        const UInt8 NUM_CORES = 7;
    54    
    55        /*!
    56         *  Interrupt vector id for Vayu/DSP.
    57         */
    58        config UInt dspIntVectId = 4;
    59    
    60        /*!
    61         *  Interrupt vector id for Vayu/EVE
    62         */
    63        config UInt eveIntVectId_INTC0 = -1;
    64        config UInt eveIntVectId_INTC1 = -1;
    65    
    66        config UInt32 procIdTable[NUM_CORES];
    67    
    68    internal:
    69    
    70        config UInt eve0ProcId      = MultiProc.INVALIDID;
    71        config UInt eve1ProcId      = MultiProc.INVALIDID;
    72        config UInt eve2ProcId      = MultiProc.INVALIDID;
    73        config UInt eve3ProcId      = MultiProc.INVALIDID;
    74        config UInt dsp0ProcId      = MultiProc.INVALIDID;
    75        config UInt dsp1ProcId      = MultiProc.INVALIDID;
    76        config UInt benelliProcId   = MultiProc.INVALIDID;
    77    }
