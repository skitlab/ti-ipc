    40    import ti.sdo.utils.MultiProc;
    41    import xdc.runtime.Error;
    42    
    43    /*!
    44     *  ======== NotifySetup ======== 
    45     *  Notify setup proxy for OMAP4430
    46     *
    47     *  This module creates and registers all drivers necessary for inter-processor
    48     *  notification on OMAP4430. 
    49     */
    50    
    51    module NotifySetup inherits ti.sdo.ipc.interfaces.INotifySetup
    52    {
    53        /*!
    54         *  Assert raised when trying to attach between OMAP4430/CORE1 and
    55         *  either the DSP or the HOST
    56         */
    57        config Error.Id E_noInterruptLine  = {
    58            msg: "E_noInterruptLine: Trying to attach between CORE1 and %s"
    59        };
    60        
    61        /*!
    62         *  Interrupt vector id for OMAP4430/DSP.
    63         */
    64        config UInt dspIntVectId = 5;
    65    
    66    internal:
    67    
    68        config UInt dspProcId       = MultiProc.INVALIDID;
    69        config UInt core0ProcId     = MultiProc.INVALIDID;
    70        config UInt core1ProcId     = MultiProc.INVALIDID;
    71        config UInt hostProcId      = MultiProc.INVALIDID;
    72    }
