    41    import xdc.runtime.Assert;
    42    
    43    /*!
    44     *  ======== GateMPSupportNull ======== 
    45     *  Module to plug GateMP proxies 
    46     */
    47    module GateMPSupportNull inherits ti.sdo.ipc.interfaces.IGateMPSupport
    48    {   
    49    
    50        /*!
    51         *  Assert raised when trying to use GateMPSupportNull's enter or leave     
    52         */
    53        config Assert.Id A_invalidAction  = {
    54            msg: "A_invalidAction: Cannot use ti.sdo.ipc.gates.GateMPSupportNull"
    55        };
    56        
    57        /*!     
    58         *  Error codes returned by certain calls in GateMP
    59         */
    60        enum Action {
    61            Action_NONE   =  0,                
    62            Action_ASSERT =  1
    63        };
    64        
    65        /*!
    66         *  ======== action ========
    67         *  Assert if the enter and/or leave is called
    68         */
    69        config Action action = Action_ASSERT;
    70        
    71    instance:    
    72    
    73        /*!
    74         *  @_nodoc
    75         *  ======== enter ========
    76         *  Enter this gate
    77         */
    78        @DirectCall
    79        override IArg enter();
    80    
    81        /*!
    82         *  @_nodoc
    83         *  ======== leave ========
    84         *  Leave this gate
    85         */
    86        @DirectCall
    87        override Void leave(IArg key);
    88    
    89    internal:
    90    
    91        struct Instance_State {
    92            UInt   resourceId;
    93        };
    94    
    95    }
