    47    interface ITransportSetup
    48    {
    49        /*! priority of the transport */
    50        config UInt priority = 0;
    51        
    52        /*! Function that will be called in MessageQ_attach */

    54        Int attach(UInt16 remoteProcId, Ptr sharedAddr);
    55    
    56        /*! Function that will be called in MessageQ_detach */
    57        @DirectCall
    58        Int detach(UInt16 remoteProcId);
    59    
    60        /*! Determines if a transport has been registered to a remote processor */
    61        @DirectCall
    62        Bool isRegistered(UInt16 remoteProcId);
    63    
    64        /*! Shared memory required */
    65        @DirectCall
    66        SizeT sharedMemReq(Ptr sharedAddr);
    67    }
