    40    import xdc.runtime.Error;
    41    
    42    /*!
    43     *  ======== INameServerRemote ========
    44     *  Interface for remote NameServer accesses
    45     */
    46    interface INameServerRemote {
    47    
    48        /*!
    49         *  ======== Status ========
    50         *  Status codes returned by certain calls
    51         */
    52        enum Status {
    53            S_SUCCESS = 0,
    54            E_FAIL = -1,
    55            E_ERROR = -2
    56        };
    57        
    58        /*! Function is called by Ipc_attach() through NameServer_SetupProxy */
    59        @DirectCall
    60        Int attach(UInt16 remoteProcId, Ptr sharedAddr);
    61    
    62        /*! Function is called by Ipc_detach() through NameServer_SetupProxy */
    63        @DirectCall
    64        Int detach(UInt16 remoteProcId);
    65    
    66        /*!
    67         *  ======== sharedMemReq ========
    68         *  Returns the shared memory size requirement for a single instance
    69         *
    70         *  Function is called during Ipc_attach() through NameServer_SetupProxy.
    71         *
    72         *  @param(sharedAddr)   Shared address
    73         *
    74         *  @b(returns)          Number of MAUs needed to create the instance.
    75         */
    76        @DirectCall
    77        SizeT sharedMemReq(Ptr sharedAddr);
    78    
    79    instance:
    80    
    81        /*!
    82         *  ======== create ========
    83         */
    84        override create(UInt16 procId);
    85    
    86        /*!
    87         *  ======== get ========
    88         */
    89        Int get(String instanceName, 
    90                String name,
    91                Ptr value,
    92                UInt32 *valueLen,
    93                xdc.runtime.knl.ISync.Handle syncHandle,
    94                Error.Block *eb);
    95    }
