    41    /*!
    42     *  ======== Build ========
    43     */
    44    
    45    @Template("./Build.xdt")
    46    metaonly module Build
    47    {
    48        
    49        /*!
    50         *  ======== libDir ========
    51         */
    52        metaonly config String libDir = null;
    53    
    54        /*!
    55         *  ======== getDefs ========
    56         *  Get the compiler -D options necessary to build
    57         */
    58        metaonly String getDefs();
    59        
    60        /*!
    61         *  ======== getCFiles ========
    62         *  Get the library C source files.
    63         */
    64        metaonly String getCFiles(String target);
    65    
    66        /*!
    67         *  ======== getAsmFiles ========
    68         *  Get the library Asm source files.
    69         */
    70        metaonly Any getAsmFiles(String target);
    71    
    72        /*
    73         *  ======== buildLibs ========
    74         *  This function generates the makefile goals for the libraries
    75         *  produced by a ti.sysbios package.
    76         */
    77        function buildLibs(objList, relList, filter, xdcArgs);
    78    
    79        /*!
    80         *  ======== getLibs ========
    81         *  Common getLibs() for all ipc packages.
    82         */
    83        function getLibs(pkg);
    84    }
